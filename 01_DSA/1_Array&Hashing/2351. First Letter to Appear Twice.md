class Solution {

    public char repeatedCharacter(String s) {

        int sLen = s.length();

        boolean[] seen = new boolean[26];

  

        for(char letter : s.toCharArray()){

            int index = letter - 'a';

  

            if(seen[index]){

                return letter;

            }

  

            seen[index] = true;

        }

        return ' ';

    }

}
