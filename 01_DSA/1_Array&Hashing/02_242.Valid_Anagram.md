class Solution {
    public int[] twoSum(int[] nums, int target) {
        int length = nums.length;
        int left = 0;
        int right = length - 1;

        while (left < right) {
            int total = nums[left] + nums[right];
            if (total == target) {
                return new int[]{left, right};
            }

            if (total > target) {
                right--;
            } else {
                left++;
            }
        }
        return new int[]{};
    }
}
