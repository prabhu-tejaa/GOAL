class Solution {

    public int longestConsecutive(int[] nums) {

        if(nums.length == 0){

            return 0;

        }

        Set<Integer> numSet = new HashSet<>();

        for(int num : nums){

            numSet.add(num);

        }

        int longestStreak = 0;

        for(int num : numSet){

            if(!numSet.contains(num - 1)){

                int currentStreak = 1;

                int nextNum = num+1;

                while(numSet.contains(nextNum)){

                    currentStreak++;

                    nextNum++;

                }

                longestStreak = Math.max(currentStreak, longestStreak);

            }

        }

        return longestStreak;

    }

}
